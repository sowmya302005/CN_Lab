#include<stdio.h>
#include<string.h>
void main()
{
    int a[20],b[50],i,j,k,n,count;
    printf("enter frame length");
    scanf("%d",&n);
    printf("enter input(0' and 1's only):");
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    i=0,count=1,j=0;
    while(i<n){
        if(a[i]==1){
            b[j]=a[i];
            for(k=i+1;a[k]==1&&k<n&&count<5;k++)
            {
                j++;
                b[j]=a[k];
                count++;
                if(count==5){
                    j++;
                    b[j]=0;
                }
                i=k;
            }
        }
        else
        {
            b[j]=a[i];
            
        }
        i++;
        j++;
    }
    printf("after stuffing frame is:");

    }
